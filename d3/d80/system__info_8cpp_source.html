48void SystemInfo::AddReplica(const ReplicaInfo& replica) {
49 if (replica.id() == 0 || replica.ip().empty() || replica.port() == 0) {
50 return;
51 }
52 for (const auto& cur_replica : replicas_) {
53 if (cur_replica.id() == replica.id()) {
54 LOG(ERROR) << " replica exist:" << replica.id();
55 return;
56 }
57 }
58 LOG(ERROR) << "add new replica:" << replica.DebugString();
59 replicas_.push_back(replica);
60}

62void SystemInfo::ProcessRequest(const SystemInfoRequest& request) {
63 switch (request.type()) {
64 case SystemInfoRequest::ADD_REPLICA: {
65 NewReplicaRequest info;
66 if (info.ParseFromString(request.request())) {
67 AddReplica(info.replica_info());
68 }
69 } break;
70 default:
71 break;
72 }
73}

124 // overrides AddReplica()
125 // Adds a replica to the least populated shard
126 // Tracks shard membership & designates first node in shard
127void SystemInfo::AddReplicaToShard(const ReplicaInfo& replica) {
128 //Check for initalization
129 if(shard_count_ == 0) {
130 LOG(ERROR) << "Set the shard count";
131 return
132 }
133
134//Find shard with smallest # of nodes
135 size_t target = 0;
136 size_t min_size = SIZE_MAX;
137
138 for(int i = 0; i < shard_count_; ++i){
139 size_t sz = shard_to_nodes[i].size();
140 if (sz < min_size){
141 target = i;
142 min_size = sz;
143 }
144 }
145//Add replica
146replicas_.push_back(replica);
147
148//Record which shard the node belongs to
149node_to_shard_[replica.id()] = target;
150
151//Add node to list
152shard_to_nodes_[target].push_back(replica.id());
153
154//Check heirarchy, and designates as primary if it is first
155if(shard_primaries_.find(target) == shard_primaries_.end()){
156 shard_primaries_[target] = replica.id();
157}
158//Log for debug
159LOG(INFO) << "Node: " << replica.id() << "Shard: " << target;
160
161 }
